\begin{frame}
\frametitle{Круговой сектор}
\begin{mytheorem}[Gongora-T.~A., Jose~J.V., Schaffner~S., Tiesinga~P.]
Собственные функции  $\psi_{k,m}(r,\varphi)$ и соответствующие уровни энергии $E_{k,m}$ для свободной частицы в круговом секторе размера  $\varphi_0$
определены формулами 
\begin{align*}
&\psi_{k, m}(r, \varphi) = J_\lambda\left(\frac{\alpha_{\lambda, k}r}{r_0}\right)\sin{\lambda\varphi}\notag \\  
&E_{k,m} = \frac{\hbar^2 \alpha_{\lambda, k}^2}{2\mu r_0^2},
 \qquad k, m \in \mathbb{N},
\end{align*}
где
$\lambda = \frac{\pi m}{\varphi_0}$ и $\alpha_{\lambda, k}$ --
$k$-ый нуль функции Бесселя  $J_\lambda(x)$  первого рода.
\end{mytheorem}
\end{frame}

\begin{frame}\frametitle{История вопроса}
В эллиптических координатах $(x,y) = (\delta \cos \varphi \cosh \rho, \delta \sin \varphi \sinh \rho)$ уравнение Шрёдингера расщепляется в систему уравнений Матьё:
\begin{equation*}
    \left\{
    \begin{aligned}
&F''(\varphi) + (\zeta-2q \cos 2\varphi) F(\varphi) = 0 && \text{(угловое)} \\
&R''(\rho) - (\zeta-2q \cosh 2\rho) R(\rho) = 0 && \text{(радиальное)}
    \end{aligned}
    \right.
\end{equation*}
Здесь  $q = \dfrac{\mu E \delta^2}{2\hbar^2}$.

%\item[1.] J. W. Strutt (Lord Rayleigh), The Theory of Sound, 1877–1878
\item[1.] É. Mathieu, Mémoire sur le mouvement vibratoire d'une membrane de forme elliptique, 1868.
\item[2.] G. Floquet, Sur les équations différentielles linéaires à coefficients périodiques, 1883.
\item[3.] N. W. McLachlan, Theory and Application of Mathieu Functions, 1947. (например)
\end{itemize}
\end{frame}

\begin{remark}
В классическом случае постоянная движения $A$ = произведение угловых моментов относительно фокусов:
$ L_{\pm \delta} = (x \pm \delta)v_y - y v_x$, тогда 

$$A = L_{-\delta} L_\delta = L_\delta L_{-\delta} = (x v_y - y v_x)^2 - \delta^2 v_y^2.$$ 

Для квантовой задачи нужны операторы угловых моментов относительно фокусов.
 $\widehat{ L_{\pm \delta} }= (x \pm \delta)\frac{\partial}{\partial y} - y\frac{\partial}{\partial x}$, тогда 
$$\widehat{A} = -\frac{1}{2}(\widehat{L_\delta} \widehat{L_{-\delta}} + \widehat{L_{-\delta}} \widehat{L_\delta} + \const \nabla ^2)$$
\end{remark}

\begin{frame}
\frametitle{Симметричная область $A_\delta$} 
\begin{mytheorem}\label{th:A}
В области $A_\delta$ собственными функциями $\psi_{k, m}(\rho, \varphi)$ оператора $\hat{H}$ являются
\begin{equation*}
\psi_{k, m}(\rho, \varphi) = 
\left[
\begin{array}{ll}
    Ce_\nu(\rho, q) ce_\nu(\varphi, q) ,   &    \text{ для нечётных $k \geq 1$}\\
    Se_\nu(\rho, q) se_\nu(\varphi, q) ,   &    \text{ для чётных $k \geq 2$}\\
\end{array}
\right.
\end{equation*}
с параметрами
$\nu = \nu_{k,m} = \nu_0+ \delta^2 \frac{\alpha_{\nu_0, m}^2}{4 r_0^2}  \nu_1 + o(\delta^2), \qquad q=q_{k,m} = \dfrac{\varkappa_{k,m}^2 \delta^2}{4},$

где
$\nu_0 = \dfrac{\pi k}{2\varphi_0}$,
%\text{\ \  and\  \  }
$\nu_1=\dfrac{\pi k \sin 2\varphi_0}{\pi^2 k^2 - 4\varphi_0^2}.$

Соответствующими собственными значениями   $\varkappa^2_{k, m}$ являются
\begin{equation*}
\varkappa^2_{k, m} = \dfrac{\alpha_{\nu_0, m}^2}{r_0^2} +
\delta^2 \dfrac{\alpha_{\nu_0, m}^3}{2 r_0^4}\dfrac{\varkappa_1 }{ \left.\frac{\partial J_{\nu_0}(u)}{\partial u}\right|_{u=\alpha_{\nu_0, m}} } + o(\delta^2),
\end{equation*}
где $\varkappa_1 = ...$
\end{mytheorem}
\end{frame}

\frametitle{Симметричная область $A_\delta$} 
\begin{mytheorem}[продолжение]

где

Здесь и далее $\alpha_{\nu_0,m}$ обозначает  $m$-ый нуль функции Бесселя  $J_{\nu_0}(x)$ первого рода.
\end{mytheorem}

Положим

$$
\widehat{A} = -\frac{1}{2}(\widehat{L_\delta} \widehat{L_{-\delta}} + \widehat{L_{-\delta}} \widehat{L_\delta}+ \delta^2 \nabla ^2),
$$
%  где  $\hat L_{\pm \delta} = (x \pm \delta)\frac{\partial}{\partial y} - y\frac{\partial}{\partial x}$
%--- операторы угловых моментов относительно фокусов.

В координатах $(x, y) = (\delta \cosh \rho \cos \varphi,  \delta \sinh \rho \sin \varphi)$, уравнение $(\nabla^2 + \varkappa^2) \psi = 0 $ расщепляется ($q = \dfrac{\delta^2 \varkappa^2}{4}$):
\begin{align*}
F''(\varphi) + (\zeta-2q \cos 2\varphi) F(\varphi) = 0, \quad \text{ решение }  F(\zeta, q, \varphi) \\
R''(\rho) - (\zeta-2q \cosh 2\rho) R(\rho) = 0, \quad \text{ решение }  R(\zeta, q, \rho)
\end{align*}

\begin{mytheorem}[C]
\label{th:L}
(a) $\hat A$ коммутирует с  $\hat H$.

(b) Пусть   $\psi(\varphi, \rho) = F(\zeta, q, \varphi) R(\zeta, q, \rho)$ --- собственная функция оператора $\hat H$. %the operator $\nabla^2$. %, ïðè÷åì $<\psi, \psi>=1$.
%Ïîëîæèì $\varphi_1(a, q, \varphi), \varphi_2(a, q, \varphi)$ -- íåçàâèñèìûå ðåøåíèÿ óðàâíåíèÿ $\frac{\partial^2}{\partial \varphi^2}\varphi + (a - 2q\cos{2\varphi})\varphi = 0$, $R_1(a, q, \rho), R_2(a, q, \rho)$ -- íåçàâèñèìûå ðåøåíèÿ óðàâíåíèÿ $\frac{\partial^2}{\partial \rho^2}R - (a - 2q\cosh{2\rho})R = 0$. 
Тогда
$\hat A \psi = \zeta \psi$
\end{mytheorem}
